# HttpClient的简单使用

导入依赖

<dependency>
    <groupId>org.apache.httpcomponents</groupId>
    <artifactId>httpclient</artifactId>
    <version>4.5.13</version>
</dependency>

# 1. GET请求

# 1.1 快速入门

public class CrawlerTest {
    public static void main(String[] args) {
        // 创建httpGet对象
        HttpGet httpGet = new HttpGet("https://www.baidu.com");
        
        try(    // 创建httpClient对象
                CloseableHttpClient httpClient = HttpClients.createDefault();
                // 发起请求,response
                CloseableHttpResponse response = httpClient.execute(httpGet)) {
            if (response.getStatusLine().getStatusCode() == 200) {
                // 获取请求体
                HttpEntity entity = response.getEntity();

                // 获取请求体内容
                String content = EntityUtils.toString(entity, "utf8");
                System.out.println(content);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

# 1.2 带参数的GET请求

# 2.2 带参数的POST请求

public class CrawlerTest {
    public static void main(String[] args) throws Exception {
        // 创建httpGet对象
        HttpPost httpPost = new HttpPost("http://yun.itheima.com/search");

        // List集合，封装表单中的数据
        List<NameValuePair> params = new ArrayList<>();

# 3. 连接池

如果每次请求都要创建HttpClient，会有频繁创建和销毁的问题，可以使用连接池来解决这个问题

public class HttpClientPoolTest {
    public static void main(String[] args) {
        // 创建连接池管理器
        PoolingHttpClientConnectionManager cm = new PoolingHttpClientConnectionManager();
        cm.setMaxTotal(100);                // 最大连接数
        cm.setDefaultMaxPerRoute(10);       // 设置每个主机的最大连接数

        // 使用连接池管理器
        doGet(cm);

    }

    private static void doGet(PoolingHttpClientConnectionManager cm) {
        CloseableHttpClient httpClient = HttpClients.custom().setConnectionManager(cm).build();     // 这个连接不需要关闭，由连接池负责管理

        HttpGet httpGet = new HttpGet("https://www.baidu.com");
        try (CloseableHttpResponse response = httpClient.execute(httpGet);) {
            if (response.getStatusLine().getStatusCode() == 200) {
                String content = EntityUtils.toString(response.getEntity(), "utf8");
                System.out.println(content);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

# 4. 请求参数

有时候因为网络，或者目标服务器的原因，请求需要更长的事件才能完成，我们需要自定义相关时间

public class CrawlerTest {
    public static void main(String[] args) throws URISyntaxException {
        // 创建httpGet对象
        HttpGet httpGet = new HttpGet("https://www.baidu.com");
        // 配置请求信息
        RequestConfig config = RequestConfig.custom()
                .setConnectTimeout(1000)            // 创建连接的最长时间 ms
                .setConnectionRequestTimeout(1000)  // 获取连接的最长时间 ms
                .setSocketTimeout(10 * 1000)        // 数据传输的最长时间 ms
                .build();

        httpGet.setConfig(config);          // 设置请求信息

        try(    // 创建httpClient对象
                CloseableHttpClient httpClient = HttpClients.createDefault();
                // 发起请求,response
                CloseableHttpResponse response = httpClient.execute(httpGet)) {
            if (response.getStatusLine().getStatusCode() == 200) {
                // 获取请求体内容
                String content = EntityUtils.toString(response.getEntity(), "utf8");
                System.out.println(content);
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
